```js
function quickSort(arr){
  return quick(arr,0,arr.length - 1);
}
function quick(arr,left,right){
  if(left < right){
    let pivot = arr[left];
    let i = left;
    let j = right;
    while(i < j){
      while(arr[j] >= pivot && i < j) j --;
      while(arr[i] <= pivot && i < j) i ++;
      if(i < j){
        let temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
      }else{
        arr[left] = arr[i];
        arr[i] = pivot;
        quick(arr,left,i - 1);
        quick(arr,i + 1,right);
        return arr;
      }
    }
  }
}
```

function lastSearch(node){
  let nodeArr = [];
  let stack = [];
  stack.push(node);
  while(stack.length){
    if(node.left && !node.touched){
      node.touched = 'left';
      node = node.left;
      stack.push(node);
      continue;
    }
    if(node.right && node.touched != 'right'){
      node.touched = 'right';
      node = node.right;
      stack.push(node);
      continue;
    }
    node = stack.pop();
    node.touched && delete node.touched;
    nodeArr.push(node.value);

function halfSearch(arr,num){
  let left = 0;
  let right = arr.length;
  while(left < right){
    let middle = Math.floor((left + right)/2);
    if(arr[middle] == num){
      return true;
    }
    if(arr[middle] > num){
      right = middle - 1;
    }
    if(arr[middle] < num){
      left = middle + 1;
    }

    return false;
  }
}

function bothwayLoopList(){
  //定义节点类
  function Node(value){
    this.value = value;
    this.pre = null;
    this.next = null;
  }
  this.head = null;
  this.tail = null;
  bothwayLoopList.prototype.add = function(value){
    let newNode = new Node(value);
    if(!this.head){
      this.head = newNode;
      this.tail = newNode;
      this.head.next = this.tail;
      this.head.pre = this.tail;
      this.tail.pre = this.head;
      this.tail.next = this.head;
    }else{
      this.tail.next = newNode;
      newNode.next = this.head;
      this.head.pre = newNode;
      newNode.pre = this.tail;
      this.tail = newNode;
    }
  }

  let memory = new Array(n);

#### 防抖和节流

###### 节流

一个函数执行一次后，只有大于设定的执行周期后，才会执行第二次

```js
function throttle(fn,delay){
  //1.记录回调函数两次执行的时间间隔
  let lastTime = 0;
  return function(){
    let nowTime = Date.now();
  	//2.如果这个时间间隔大于时间限制，那么就执行函数
  	if(nowTime - lastTime > delay){
    	fn();
      lastTime = nowTime;
  	}
  };
}
```

###### 防抖

一个会频繁触发的函数，在规定的时间内，只让最后一次生效，前面的不生效（防止重复操作）

```js
//在事件被触发n秒后再执行回调，如果在这n秒内又被触发，则重新计时
function debounce(fn,delay){
  let timer = null;
  return function(){
    clearTimeout(timer);
    timer = setTimeout(fn,delay);
  };
};
```

function ajax(url){
  return new Promise((resolve,reject) => {
    let xhr = new XMLHttpRequest();
